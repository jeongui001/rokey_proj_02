# stt_node 음성 명령 인식(웨이크워드 + OpenAI STT)에 필요한 pip 의존성.
openwakeword     # 로컬/무료 웨이크워드 감지("헬로 로키", resource/hello_rokey_8332_32.tflite)

numpy            # 오디오 버퍼 처리
openai           # 녹음된 명령을 텍스트로 변환(whisper-1 API)
python-dotenv    # resource/.env에서 OPENAI_API_KEY 로드
